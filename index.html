My things for a happy life


def find_minimum_things_needed_for_my_happy_life(lifetime=90):

	important_things = {"family":60, "money":50, "good_work":20, "pets/music":10}

	importance = [v for v in important_things.values()]

	req_things = [0 for i in range(0,lifetime+1)]

	for life in range(1,lifetime+1):

		req_values = [-1 for j in range(len(importance))]

		for ix,value in enumerate(importance): 
			if life >= value:
				req_values[ix] = req_things[life - value] + 1

		for j in req_values:
			if j != -1 and (req_things[life] == 0 or req_things[life] > j):
				req_things[life] = j

	return req_things[lifetime] 
    
if __name__ == '__main__':
	print("Minimum things required : ", find_minimum_things_needed_for_my_happy_life())
